cmake_minimum_required (VERSION 3.20)
project (${TICOS_PROJ_NAME})

# compiling flags
set (CMAKE_C_STANDARD 11)
add_definitions (-Wall -Werror -Wextra -Wstrict-prototypes)

if (DEFINED ENV{TICOS_PATH})
set (TICOS_PATH $ENV{TICOS_PATH})
else ()
set (TICOS_PATH ${CMAKE_SOURCE_DIR}/../../..)
endif ()
set (TICOS_PLATFORM unix)

include (${TICOS_PATH}/scripts/cmake/ticos.cmake)

aux_source_directory (${CMAKE_SOURCE_DIR} PROJECT_SRCS)
include_directories (${TICOS_INCS})
add_executable (${PROJECT_NAME} ${FRAMEWORK_SRCS} ${PROJECT_SRCS})
